<script>
(async()=>{
const video=document.getElementById('video');
const drawCanvas=document.getElementById('drawCanvas');
const uiCanvas=document.getElementById('uiCanvas');
const drawCtx=drawCanvas.getContext('2d');
const uiCtx=uiCanvas.getContext('2d');

function resize(){
 drawCanvas.width=uiCanvas.width=innerWidth;
 drawCanvas.height=uiCanvas.height=innerHeight;
}
resize(); addEventListener('resize',resize);

/* ===== COLOR STATE ===== */
let currentColor='red';
const colorEls=[...document.querySelectorAll('.color')];
const clearBtn=document.getElementById('clearBtn');

/* ===== SMOOTH ===== */
const SMOOTH=0.6;
let smooth=null;
const lerp=(a,b,t)=>a+(b-a)*t;
function smoothPoints(src){
 if(!smooth){smooth=src.map(p=>({...p}));return smooth}
 src.forEach((p,i)=>{
  smooth[i].x=lerp(smooth[i].x,p.x,SMOOTH);
  smooth[i].y=lerp(smooth[i].y,p.y,SMOOTH);
 });
 return smooth;
}

/* ===== HANDS ===== */
const hands=new Hands({
 locateFile:f=>`https://cdn.jsdelivr.net/npm/@mediapipe/hands/${f}`
});
hands.setOptions({
 maxNumHands:1,
 modelComplexity:1,
 minDetectionConfidence:0.7,
 minTrackingConfidence:0.7
});

/* ===== DRAW STATE ===== */
let drawing=false;
let lastX=null,lastY=null;
let wasPinch=false;

hands.onResults(res=>{
 uiCtx.clearRect(0,0,uiCanvas.width,uiCanvas.height);

 if(!res.multiHandLandmarks?.length){
  drawing=false; smooth=null; wasPinch=false;
  return;
 }

 const lm=smoothPoints(res.multiHandLandmarks[0]);

 const i=lm[8], t=lm[4];
 const dist=Math.hypot(i.x-t.x,i.y-t.y);
 const pinch=dist<0.045;

 const cx=(i.x+t.x)/2;
 const cy=(i.y+t.y)/2;
 const px=(1-cx)*drawCanvas.width;
 const py=cy*drawCanvas.height;

 /* ===== UI HIT TEST ===== */
 let hoverColor=null;
 colorEls.forEach(el=>{
  el.classList.remove('hover');
  const r=el.getBoundingClientRect();
  if(px>r.left&&px<r.right&&py>r.top&&py<r.bottom){
   el.classList.add('hover');
   hoverColor=el;
  }
 });

 clearBtn.classList.remove('hover');
 const cr=clearBtn.getBoundingClientRect();
 const hoverClear=px>cr.left&&px<cr.right&&py>cr.top&&py<cr.bottom;
 if(hoverClear) clearBtn.classList.add('hover');

 /* ===== APPLY ON RELEASE ===== */
 if(wasPinch && !pinch){
  if(hoverColor){
   currentColor=hoverColor.dataset.color;
   colorEls.forEach(c=>c.classList.remove('active'));
   hoverColor.classList.add('active');
  }
  if(hoverClear){
   drawCtx.clearRect(0,0,drawCanvas.width,drawCanvas.height);
  }
 }

 /* ===== DRAW ===== */
 if(pinch && !hoverColor && !hoverClear){
  if(!drawing){
   drawing=true; lastX=px; lastY=py;
  }else{
   drawCtx.strokeStyle=currentColor;
   drawCtx.lineWidth=5;
   drawCtx.lineCap='round';
   drawCtx.beginPath();
   drawCtx.moveTo(lastX,lastY);
   drawCtx.lineTo(px,py);
   drawCtx.stroke();
   lastX=px; lastY=py;
  }
 }else{
  drawing=false;
 }

 /* ===== CURSOR ===== */
 uiCtx.fillStyle=pinch?'lime':'white';
 uiCtx.beginPath();
 uiCtx.arc(px,py,pinch?10:6,0,Math.PI*2);
 uiCtx.fill();

 wasPinch=pinch;
});

/* ===== CAMERA ===== */
const stream=await navigator.mediaDevices.getUserMedia({
 video:{facingMode:'user'},audio:false
});
video.srcObject=stream;
await video.play();

async function loop(){
 await hands.send({image:video});
 requestAnimationFrame(loop);
}
loop();
})();
</script>
